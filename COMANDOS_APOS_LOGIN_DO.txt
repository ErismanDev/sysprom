==========================================
COMANDOS PARA EXECUTAR APÓS LOGIN NO SERVIDOR
==========================================

# 1. Ir para o diretório do projeto
cd /home/seprom/sepromcbmepi

# 2. Verificar se está no diretório correto
pwd
# Deve mostrar: /home/seprom/sepromcbmepi

# 3. Fazer backup dos arquivos atuais
cp static/js/chat-widget-ios.js static/js/chat-widget-ios.js.backup
cp static/js/chat-tempo-real.js static/js/chat-tempo-real.js.backup
cp static/js/chat-calls.js static/js/chat-calls.js.backup

# 4. Verificar se os arquivos existem
ls -lh static/js/chat-*.js

# 5. AGORA você precisa fazer upload dos arquivos atualizados
#    Use WinSCP ou SCP de outro terminal para fazer upload
#    OU edite os arquivos manualmente com nano

# 6. Após fazer upload/editar os arquivos, execute:

# Ativar ambiente virtual
source venv/bin/activate

# Coletar arquivos estáticos
python manage.py collectstatic --noinput

# Reiniciar Gunicorn
sudo systemctl restart gunicorn

# Verificar status
sudo systemctl status gunicorn

# Se tudo estiver OK, você verá: "Active: active (running)"

# 7. Verificar logs se houver erro
sudo journalctl -u gunicorn -n 50 --no-pager

# 8. Sair do servidor
exit
