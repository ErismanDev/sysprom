==========================================
COMANDOS PARA VERIFICAR E REINICIAR GUNICORN
==========================================

# 1. Verificar se o gunicorn está rodando
ps aux | grep gunicorn

# 2. Verificar serviços systemd relacionados
sudo systemctl list-units | grep -i gunicorn
sudo systemctl list-units | grep -i seprom
sudo systemctl list-units | grep -i django

# 3. Verificar se há um serviço com outro nome
sudo systemctl list-units --type=service | grep -E "(web|app|django|seprom)"

# 4. Verificar se está usando supervisor
sudo supervisorctl status

# 5. Verificar processos Python rodando
ps aux | grep python | grep manage

# 6. Verificar se há um script de inicialização
ls -la /etc/systemd/system/ | grep -E "(gunicorn|seprom|django)"
ls -la /etc/supervisor/conf.d/ | grep -E "(gunicorn|seprom|django)"

# 7. Se encontrar o serviço, reiniciar:
#    sudo systemctl restart NOME_DO_SERVICO

# 8. Se estiver usando supervisor:
#    sudo supervisorctl restart NOME_DO_PROCESSO

# 9. Se não houver serviço, pode estar rodando manualmente
#    Nesse caso, apenas recarregar a página já deve funcionar
#    (os arquivos estáticos foram coletados)

# 10. Verificar se o Nginx precisa ser reiniciado
sudo systemctl status nginx
sudo systemctl restart nginx
